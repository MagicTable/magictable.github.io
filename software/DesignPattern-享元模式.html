DesignPattern - дә«е…ғжЁЎејҸ

йӯ”жңҜжЎҢ2024е№ҙ11жңҲ18ж—ҘиҪҜд»¶зЁӢеәҸ/и®ҫи®ЎжЁЎејҸиҝӣеәҰ-е·Іе®ҢжҲҗзҹҘиҜҶжҖ»з»“

ж–Үз« ж‘ҳиҰҒпјҡ дә«е…ғжЁЎејҸдҪҝз”ЁжҖ»з»“гҖӮ

з®Җд»Ӣ

з®ҖиҰҒиҜҙжҳҺ

  • дә«е…ғжЁЎејҸпјҲFlyweight PatternпјүжҳҜдёҖз§Қз»“жһ„еһӢи®ҫи®ЎжЁЎејҸгҖӮ
  • йҖҡиҝҮе…ұдә«е°ҪеҸҜиғҪеӨҡзҡ„зӣёдјјеҜ№иұЎжқҘеҮҸе°‘еҶ…еӯҳзҡ„дҪҝз”ЁпјҢжҸҗй«ҳзЁӢеәҸзҡ„жҖ§иғҪгҖӮ

дё»иҰҒеҠҹиғҪ

  • еҮҸе°‘еҲӣе»әеҜ№иұЎзҡ„ж•°йҮҸпјҢеҮҸе°‘еҶ…еӯҳеҚ з”Ёе’ҢжҸҗй«ҳжҖ§иғҪгҖӮ
  • жҸҗдҫӣе…ұдә«зҡ„жҺҘеҸЈпјҢдҪҝеҫ—зӣёеҗҢзҡ„еҜ№иұЎеҸҜд»Ҙиў«еӨҚз”ЁгҖӮ

жіЁж„ҸдәӢйЎ№

  • зЎ®дҝқеҜ№иұЎжҳҜдёҚеҸҜеҸҳеҜ№иұЎпјҢд»ҘдҝқиҜҒе…ұдә«зҡ„е®үе…ЁжҖ§гҖӮ
  • еҲҶзҰ»еҶ…йғЁзҠ¶жҖҒпјҲеӣәжңүеұһжҖ§пјүе’ҢеӨ–йғЁзҠ¶жҖҒпјҲеҸҜеҸҳеұһжҖ§пјүпјҢеҶ…йғЁзҠ¶жҖҒеҸҜд»Ҙиў«е…ұдә«пјҢеӨ–йғЁзҠ¶жҖҒеҲҷдёҚеҸҜд»ҘгҖӮ
  • йңҖиҰҒз®ЎзҗҶеҜ№иұЎжұ пјҢд»Ҙе®һзҺ°еҜ№иұЎзҡ„е…ұдә«гҖӮ

йҖӮз”ЁеңәжҷҜ

  • еҪ“дёҖдёӘзЁӢеәҸдёӯдҪҝз”ЁдәҶеӨ§йҮҸзҡ„еҜ№иұЎгҖӮ
  • иҝҷдәӣеҜ№иұЎйҖ жҲҗдәҶеҫҲеӨ§зҡ„еҶ…еӯҳејҖй”ҖгҖӮ
  • иҝҷдәӣеҜ№иұЎзҡ„еӨ§еӨҡж•°зҠ¶жҖҒеҸҜд»ҘеӨ–йғЁеҢ–гҖӮ
  • иҝҷдәӣеҜ№иұЎжҳҜдёҚеҸҜеҸҳзҡ„гҖӮ

жЎҲдҫӢ

import java.util.HashMap;
import java.util.Map;

// дә«е…ғжҺҘеҸЈпјҢе®ҡд№үдәҶдә«е…ғеҜ№иұЎйңҖиҰҒе®һзҺ°зҡ„ж“ҚдҪң
interface Flyweight {
    void operation(String extrinsicState);
}

// е…·дҪ“зҡ„дә«е…ғзұ»пјҢе®һзҺ°дәҶдә«е…ғжҺҘеҸЈ
class ConcreteFlyweight implements Flyweight {
    // еҶ…йғЁзҠ¶жҖҒпјҢеҚідә«е…ғеҜ№иұЎзҡ„еӣәжңүеұһжҖ§пјҢеҸҜд»Ҙиў«е…ұдә«
    private String intrinsicState;

    // жһ„йҖ ж–№жі•пјҢеҲқе§ӢеҢ–еҶ…йғЁзҠ¶жҖҒ
    public ConcreteFlyweight(String intrinsicState) {
        this.intrinsicState = intrinsicState;
    }

    // е®һзҺ°operationж–№жі•пјҢдҪҝз”ЁеӨ–йғЁзҠ¶жҖҒ
    @Override
    public void operation(String extrinsicState) {
        System.out.println("Intrinsic State = " + this.intrinsicState);
        System.out.println("Extrinsic State = " + extrinsicState);
    }
}

// дә«е…ғе·ҘеҺӮпјҢиҙҹиҙЈеҲӣе»әе’Ңз®ЎзҗҶдә«е…ғеҜ№иұЎ
class FlyweightFactory {
    // дә«е…ғжұ пјҢз”ЁдәҺеӯҳеӮЁдә«е…ғеҜ№иұЎпјҢд»ҘдҫҝеӨҚз”Ё
    private Map<String, Flyweight> flyweights = new HashMap<>();

    // иҺ·еҸ–дә«е…ғеҜ№иұЎзҡ„ж–№жі•
    public Flyweight getFlyweight(String key) {
        // еҰӮжһңдә«е…ғжұ дёӯеӯҳеңЁиҜҘеҜ№иұЎпјҢеҲҷзӣҙжҺҘиҝ”еӣһ
        if (flyweights.containsKey(key)) {
            return flyweights.get(key);
        } else {
            // еҰӮжһңдә«е…ғжұ дёӯдёҚеӯҳеңЁиҜҘеҜ№иұЎпјҢеҲҷеҲӣе»әдёҖдёӘж–°зҡ„дә«е…ғеҜ№иұЎпјҢе№¶ж”ҫе…Ҙдә«е…ғжұ дёӯ
            Flyweight flyweight = new ConcreteFlyweight(key);
            flyweights.put(key, flyweight);
            return flyweight;
        }
    }
}

// е®ўжҲ·з«Ҝд»Јз ҒпјҢдҪҝз”Ёдә«е…ғжЁЎејҸ
public class FlyweightPatternDemo {
    public static void main(String[] args) {
        // еҲӣе»әдә«е…ғе·ҘеҺӮ
        FlyweightFactory factory = new FlyweightFactory();

        // йҖҡиҝҮдә«е…ғе·ҘеҺӮиҺ·еҸ–дә«е…ғеҜ№иұЎ
        Flyweight flyweight1 = factory.getFlyweight("key1");
        Flyweight flyweight2 = factory.getFlyweight("key2");

        // ж“ҚдҪңдә«е…ғеҜ№иұЎпјҢдј е…ҘеӨ–йғЁзҠ¶жҖҒ
        flyweight1.operation("Extrinsic State A");
        flyweight2.operation("Extrinsic State B");

        // еҶҚж¬ЎиҺ·еҸ–дә«е…ғеҜ№иұЎпјҢжӯӨж—¶е°ҶеӨҚз”Ёе·ІеӯҳеңЁзҡ„дә«е…ғеҜ№иұЎ
        Flyweight flyweight3 = factory.getFlyweight("key1");
        flyweight3.operation("Extrinsic State C");

        // жЈҖжҹҘflyweight1е’Ңflyweight3жҳҜеҗҰдёәеҗҢдёҖдёӘеҜ№иұЎ
        System.out.println(flyweight1 == flyweight3); // иҫ“еҮәtrue
    }
}
