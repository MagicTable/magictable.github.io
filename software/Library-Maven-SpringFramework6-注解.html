    public void logon(@RequestParam("id") String id, @RequestParam("name") String name) {
    }

    // URLпјҡhttp://127.0.0.1:8080/web/logon/123/е°ҸжҳҺ
    // жүӢеҠЁжҢҮе®ҡиҜ·жұӮurlз»‘е®ҡеҲ°иҜҘж–№жі•пјҢйҖҡиҝҮ`{}`еҚ дҪҚз¬Ұз»‘е®ҡ

    public void login(@PathVariable("id") String id, @PathVariable("name") String name) {
    }
}

@Value

  • жіЁе…Ҙеҹәжң¬зұ»еһӢпјҢдёҖиҲ¬з”ЁдәҺиҜ»еҸ–й…ҚзҪ®ж–Үд»¶гҖҒжүӢеҠЁдј е…Ҙе…·дҪ“еҸӮж•°гҖӮдҫӢеҰӮпјҡ@Value("е°ҸжҳҺ")гҖҒ@Value("${mysql.password}")

@Required

  • йҖӮз”ЁдәҺbeanеұһжҖ§setterж–№жі•жҳҫејҸиөӢеҖјжҲ–иҮӘеҠЁжіЁе…ҘеҖјгҖӮ
